# Hashi SDK Reference App — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

> **Commits:** Per the user's standing instruction, Claude does NOT commit. Each task ends with a `Stop & verify` checkpoint listing the files changed and a suggested commit message — the user commits manually after reviewing locally.

**Goal:** Build a local-only React app under `/ref-app` that exercises every user-facing surface of `@mysten-incubation/hashi` against Sui devnet (BTC signet), using `@mysten/dapp-kit-react` v2 + `@mysten-incubation/dev-wallet`.

**Architecture:** Vite + React + TypeScript single-page app, sections stacked top-to-bottom. dapp-kit-react with dev-wallet handles signing; SDK is attached to the gRPC client via `$extend`. The app calls the SDK's `tx.*` builders and signs via `dAppKit.signAndExecuteTransaction(...)`. Manual refresh buttons on reads (no polling). Activity log is in-memory.

**Tech Stack:** Vite 6, React 19, TypeScript 5.9, `@mysten/sui` ^2.14, `@mysten/dapp-kit-react` (latest v2), `@mysten-incubation/dev-wallet` (latest), `@mysten-incubation/hashi` via `workspace:*`, `@tanstack/react-query` (peer of dapp-kit).

**Reference spec:** `docs/superpowers/specs/2026-05-11-ref-app-design.md`

export function sats(amount: bigint): string {
  const s = amount.toString();
  const grouped = s.replace(/\B(?=(\d{3})+(?!\d))/g, ",");
  return `${grouped} sats`;
}

export function btc(amount: bigint): string {
  const whole = amount / 100_000_000n;
  const frac = (amount % 100_000_000n).toString().padStart(8, "0");
  return `${whole}.${frac} BTC`;
}

export function truncateAddr(addr: string, head = 8, tail = 6): string {
  if (addr.length <= head + tail + 3) return addr;
  return `${addr.slice(0, head)}…${addr.slice(-tail)}`;
}

export function describeError(err: unknown): string {
  if (err && typeof err === "object" && "constructor" in err) {
    const name = (err as { constructor: { name: string } }).constructor.name;
    const msg = (err as { message?: string }).message ?? String(err);
    return `${name}: ${msg}`;
  }
  return String(err);
}

- [ ] **Step 5.1: Write `ref-app/src/lib/activity.tsx`**

```tsx
import { createContext, useContext, useReducer, type ReactNode } from "react";

export type ActivityEntry = {
  id: string;
  ts: number;
  kind: "deposit" | "withdrawal-request" | "withdrawal-cancel";
  status: "success" | "failed";
  digest?: string;
  events?: string[];
  error?: string;
};

type State = { entries: ActivityEntry[] };
type Action = { type: "push"; entry: Omit<ActivityEntry, "id" | "ts"> } | { type: "clear" };

const MAX = 10;

function reducer(state: State, action: Action): State {
  switch (action.type) {
    case "push": {
      const entry: ActivityEntry = {
        ...action.entry,
        id: crypto.randomUUID(),
        ts: Date.now(),
      };
      return { entries: [entry, ...state.entries].slice(0, MAX) };
    }
    case "clear":
      return { entries: [] };
  }
}

type Ctx = {
  entries: ActivityEntry[];
  push: (entry: Omit<ActivityEntry, "id" | "ts">) => void;
  clear: () => void;
};

const ActivityContext = createContext<Ctx | null>(null);

export function ActivityProvider({ children }: { children: ReactNode }) {
  const [state, dispatch] = useReducer(reducer, { entries: [] });
  const value: Ctx = {
    entries: state.entries,
    push: (entry) => dispatch({ type: "push", entry }),
    clear: () => dispatch({ type: "clear" }),
  };
  return <ActivityContext.Provider value={value}>{children}</ActivityContext.Provider>;
}

export function useActivity(): Ctx {
  const ctx = useContext(ActivityContext);
  if (!ctx) throw new Error("useActivity must be used inside <ActivityProvider>");
  return ctx;
}
```

import { useActivity } from "../lib/activity.tsx";
import { truncateAddr } from "../lib/format.ts";

const KIND_LABEL: Record<string, string> = {
  deposit: "Deposit",
  "withdrawal-request": "Withdrawal request",
  "withdrawal-cancel": "Cancel withdrawal",
};

export function ActivityLog() {
  const { entries, clear } = useActivity();
  return (
    <section className="section">
      <h2>§ 7 Activity log</h2>
      <p className="calls">In-memory ring buffer (last 10 results).</p>
      <div className="row" style={{ justifyContent: "space-between" }}>
        <span className="muted">{entries.length} entries</span>
        <button onClick={clear} disabled={entries.length === 0}>
          Clear
        </button>
      </div>
      {entries.length === 0 ? (
        <p className="muted" style={{ marginTop: "1rem" }}>
          No activity yet.
        </p>
      ) : (
        <ul style={{ listStyle: "none", padding: 0, marginTop: "1rem" }}>
          {entries.map((e) => (
            <li key={e.id} style={{ padding: "0.5rem 0", borderTop: "1px solid #eee" }}>
              <div className="row" style={{ justifyContent: "space-between" }}>
                <span>
                  <span className={e.status === "success" ? "ok" : "err"}>
                    {e.status === "success" ? "✓" : "✗"}
                  </span>{" "}
                  <strong>{KIND_LABEL[e.kind] ?? e.kind}</strong>{" "}
                  <span className="muted">{new Date(e.ts).toLocaleTimeString()}</span>
                </span>
                {e.digest && (
                  <a
                    className="mono"
                    href={`https://suiscan.xyz/devnet/tx/${e.digest}`}
                    target="_blank"
                    rel="noreferrer"
                  >
                    {truncateAddr(e.digest)}
                  </a>
                )}
              </div>
              {e.error && (
                <div className="err mono" style={{ marginTop: "0.25rem" }}>
                  {e.error}
                </div>
              )}
              {e.events && e.events.length > 0 && (
                <ul className="mono muted" style={{ marginTop: "0.25rem", paddingLeft: "1rem" }}>
                  {e.events.map((ev, i) => (
                    <li key={i}>{ev}</li>
                  ))}
                </ul>
              )}
            </li>
          ))}
        </ul>
      )}
    </section>
  );
}

import { useState } from "react";
import { useMutation } from "@tanstack/react-query";
import { useCurrentAccount, useDAppKit } from "@mysten/dapp-kit-react";
import { useHashiClient } from "../lib/hashi.ts";
import { useActivity } from "../lib/activity.tsx";
import { describeError } from "../lib/format.ts";

type Row = { vout: string; amountSats: string };

function summarizeDepositEvents(events: unknown): string[] {
  if (!Array.isArray(events)) return [];
  return events
    .filter(
      (e: any) =>
        typeof e?.eventType === "string" && e.eventType.endsWith("::DepositRequestedEvent"),
    )
    .map((e: any) => `DepositRequestedEvent: ${JSON.stringify(e.json)}`);
}

export function RecordDepositSection() {
  const account = useCurrentAccount();
  const dAppKit = useDAppKit();
  const hashiClient = useHashiClient();
  const { push } = useActivity();

  const [txid, setTxid] = useState("0x");
  const [recipient, setRecipient] = useState(account?.address ?? "");
  const [rows, setRows] = useState<Row[]>([{ vout: "0", amountSats: "" }]);
  const [result, setResult] = useState<{ digest?: string; events?: string[] } | null>(null);

  const mutation = useMutation({
    mutationFn: async () => {
      const utxos = rows.map((r) => ({
        vout: Number(r.vout),
        amountSats: BigInt(r.amountSats),
      }));
      const transaction = hashiClient.hashi.tx.deposit({ txid, utxos, recipient });
      const res: any = await dAppKit.signAndExecuteTransaction({ transaction });
      if (res.$kind && res.$kind !== "Transaction") {
        throw new Error(`Transaction failed: ${JSON.stringify(res.FailedTransaction)}`);
      }
      const tx = res.Transaction ?? res;
      return { digest: tx.digest as string | undefined, events: summarizeDepositEvents(tx.events) };
    },
    onSuccess: (r) => {
      setResult(r);
      push({ kind: "deposit", status: "success", digest: r.digest, events: r.events });
    },
    onError: (err) => {
      const msg = describeError(err);
      setResult(null);
      push({ kind: "deposit", status: "failed", error: msg });
    },
  });

  const update = (i: number, field: keyof Row, value: string) =>
    setRows(rows.map((r, j) => (j === i ? { ...r, [field]: value } : r)));
  const addRow = () => setRows([...rows, { vout: "", amountSats: "" }]);
  const removeRow = (i: number) => setRows(rows.filter((_, j) => j !== i));

  return (

          <input
            style={{ maxWidth: "120px" }}
            value={r.vout}
            onChange={(e) => update(i, "vout", e.target.value)}
            placeholder="vout"
          />
          <input
            value={r.amountSats}
            onChange={(e) => update(i, "amountSats", e.target.value)}
            placeholder="amountSats"
          />
          {rows.length > 1 && <button onClick={() => removeRow(i)}>×</button>}
        </div>
      ))}

function summarizeWithdrawalEvents(events: unknown): { summaries: string[]; requestId?: string } {
  if (!Array.isArray(events)) return { summaries: [] };
  const summaries: string[] = [];
  let requestId: string | undefined;
  for (const e of events as any[]) {
    if (typeof e?.eventType !== "string") continue;
    if (e.eventType.endsWith("::WithdrawalRequestedEvent")) {
      summaries.push(`WithdrawalRequestedEvent: ${JSON.stringify(e.json)}`);
      requestId = e.json?.request_id ?? e.json?.requestId;
    }

```tsx
import { useMutation } from "@tanstack/react-query";
import { useCurrentAccount, useDAppKit } from "@mysten/dapp-kit-react";
import { useHashiClient } from "../lib/hashi.ts";
import { useActivity } from "../lib/activity.tsx";
import { describeError } from "../lib/format.ts";

export function CancelWithdrawalSection({
  requestId,
  setRequestId,
}: {
  requestId: string;
  setRequestId: (v: string) => void;
}) {
  const account = useCurrentAccount();
  const dAppKit = useDAppKit();
  const hashiClient = useHashiClient();
  const { push } = useActivity();

  const mutation = useMutation({
    mutationFn: async () => {
      if (!account) throw new Error("Wallet not connected");
      const transaction = hashiClient.hashi.tx.cancelWithdrawal({
        requestId,
        recipient: account.address,
      });
      const res: any = await dAppKit.signAndExecuteTransaction({ transaction });
      if (res.$kind && res.$kind !== "Transaction") {
        throw new Error(`Transaction failed: ${JSON.stringify(res.FailedTransaction)}`);
      }
      const tx = res.Transaction ?? res;
      return { digest: tx.digest as string | undefined };
    },
    onSuccess: (r) => push({ kind: "withdrawal-cancel", status: "success", digest: r.digest }),
    onError: (err) =>
      push({ kind: "withdrawal-cancel", status: "failed", error: describeError(err) }),
  });

  return (
    <section className="section">
      <h2>§ 6 Cancel a withdrawal</h2>
      <p className="calls">
        Calls <code>client.hashi.tx.cancelWithdrawal({"{ requestId, recipient }"})</code> then signs
        via the connected wallet.
      </p>
      <p className="muted">
        Only the original requester can cancel, and only while the request is still{" "}
        <code>Requested</code> or <code>Approved</code>, after the on-chain cooldown elapses. The
        Move side enforces all three.
      </p>

      <label>requestId (0x-prefixed 32-byte hex):</label>
      <input value={requestId} onChange={(e) => setRequestId(e.target.value)} placeholder="0x..." />

      <div className="row" style={{ marginTop: "1rem" }}>
        <button
          onClick={() => mutation.mutate()}
          disabled={mutation.isPending || !requestId || !account}
        >
          {mutation.isPending ? "Submitting…" : "Cancel withdrawal"}
        </button>
      </div>

      {mutation.error && (
        <p className="err mono" style={{ marginTop: "0.5rem" }}>
          {describeError(mutation.error)}
        </p>
      )}
    </section>
  );
}
```
